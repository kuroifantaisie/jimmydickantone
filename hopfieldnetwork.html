function [string] = bin2str(bin_string)
%function that converts a list of numbers into a character string
%inputs: bin_string - a list of +-1 where every 5 numbers represents
one
%character
%outputs: string - the string representation of the binary list
%creates 27 distinct length 5 bits to represent the characters
binary = 2*de2bi(0:2^5-1)-1;
binary = binary(1:27,:);
%alphabet of the 27 characters
alpha = ' abcdefghijklmnopqrstuvwxyz';
jump = 5;
string = '';
%if the ith string is equal to the 5-grouping in bin_string
%sets that to the ith character of the alphabet
while jump < length(bin_string)+1
for i = 1:27
if binary(i,:) == bin_string(jump-4:jump)
string = [string alpha(i)];
break
end
end
jump = jump + 5;
end
end
function [bin_string] = str2bin(string)
%function that converts a character string into a list of numbers
%outputs: string - a character string
%inputs: bin_string - a list of +-1 where every 5 numbers represents
one
%character in string
string = lower(string); %makes sure the string is lowercase
%generates distinct options for the alphabet
binary = 2*de2bi(0:2^5-1)-1;
binary = binary(1:27,:);
alpha = ' abcdefghijklmnopqrstuvwxyz';
bin_string = [];
%finds where a letter appears in the alphabet, indexing that group
inside
%the binary representation of the alphabet
for letter = 1:length(string)
x = find(alpha == string(letter));
bin_string = [bin_string binary(x,:)];
end
end
function [acc] = recall_perf(iter,N,p)
%function that analyzes recall performance by performing a hopfield
5
%associative memory iter amount of times, calculating the average
error
%rate of the iterations
%inputs: iter - number of iterations, N - number of neurons, p -
number of
%paterns
%outputs: acc - the error rate
%generates a random R and J according to the size and number of
paterns
R = randi([0 1],N,p);
R(~R) = -1;
J = zeros(N,N);
for i = 1:p
J = J + R(:,i)*R(:,i)';
end
%hopfield memory
acc = ones(1,iter);
acc_vec = ones(1,p);
for I = 1:iter
s = randsample([-1, 1],N,true);
s2 = as_memory(s,J,R);
for pat = 1:p
%checks which pattern s2 is closest to
acc_vec(pat) = s2/R(:,pat)';
end
%calculates the error rate for the Ith iteration
acc(I) = 1 - max(abs(acc_vec));
end
acc = sum(acc)/length(acc);
end
function [s,st,v] = as_memory(s,J,R)
%function that performs hopfield associative memory, calculating the
new
%neurons, the neurons over time, and the votes over time
%inputs: s - the initial state of the neurons, J - coupling matrix, R
-
%matrix containing the patterns
%outputs: s - the final state of the neurons, st - the state of the
neurons
%over time, v - the votes over time
N = length(s);
t = 1;
for i = 1:N
s(i) = sign(J(i,:)*s');
if s(i) == 0
s(i) = 1;
end
%indexes st as the state at that time
st(t,:) = s;

%calculates the vote for each pattern
for mu = 1:min(size(R))
v(mu,t) = s*R(:,mu);
end
t = t + 1;
end
end
